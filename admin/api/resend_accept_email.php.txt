declare(strict_types=1);

header('Content-Type: application/json; charset=utf-8');

require_once __DIR__ . '/send_accept_link.php';
require_once __DIR__ . '/../../common/lib/datetime_fmt.php';

$raw = file_get_contents('php://input') ?: '';
$body = json_decode($raw, true);

$id = '';
if (isset($_POST['id'])) {
    $id = trim((string)$_POST['id']);
} elseif (is_array($body) && isset($body['id'])) {
    $id = trim((string)$body['id']);
}

if ($id === '') {
    http_response_code(400);
    echo json_encode(['ok' => false, 'error' => 'missing_id']);
    exit;
}

try {
    $res = cm_send_accept_link($id, true); // drugi parameter: force-resend
    $ok  = (bool)($res['ok'] ?? false);
    $err = $res['error'] ?? null;

    echo json_encode([
        'ok'         => $ok,
        'mail_error' => $err,
        'res'        => $res
    ]);
} catch (Throwable $e) {
    echo json_encode([
        'ok' => false,
        'error' => $e->getMessage()
    ]);
}
